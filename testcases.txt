->

  a,
  b

# subquery
select * from (select a, b from d) x
->

  *
FROM
  (SELECT
    a,
    b
  FROM
    d) x

# with
with a as (select * from x), b as (select * from y) select * from
a inner join b on a.id = b.id
->
WITH
---
a AS
(SELECT
  *
FROM
  x),
---
b AS
(SELECT
  *
FROM
  y)
---
SELECT
  *
FROM
  a
  INNER JOIN
  b
